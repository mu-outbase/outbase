Core08-A3 反映手順

1. ZIPをGitHub mainへアップロード
   https://github.com/mu-outbase/outbase/upload/main

2. アップロードZIP
   OUTBASE_Core08_A3_DATA_GUARD_完成版.zip

3. workflowファイルを作成
   https://github.com/mu-outbase/outbase/new/main

4. ファイル名
   .github/workflows/unpack-core08-a3.yml

5. workflow内容

name: Unpack Core08-A3 ZIP

on:
  push:
    paths:
      - 'OUTBASE_Core08_A3_DATA_GUARD_完成版.zip'
      - '.github/workflows/unpack-core08-a3.yml'
  workflow_dispatch:

permissions:
  contents: write

jobs:
  unpack:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4
      - name: Unzip Core08-A3 package
        run: |
          unzip -o "OUTBASE_Core08_A3_DATA_GUARD_完成版.zip" -d .
          rm -f "OUTBASE_Core08_A3_DATA_GUARD_完成版.zip"
      - name: Commit unpacked files
        run: |
          git config user.name "github-actions"
          git config user.email "github-actions@github.com"
          git add .
          git commit -m "Core08-A3 data guard foundation" || echo "No changes to commit"
          git push
